单击版

#include<cstdio>   
#include<windows.h>    
#define KEY_DOWN(VK_NONAME) ((GetAsyncKeyState(VK_NONAME) & 0x8000) ? 1:0)   
int main()
{ 
    int now=0,k=0;   
    printf("长按1将鼠标放在“采集猫薄荷”上\n");  
    while(1){    
        if(KEY_DOWN('1')){   
            mouse_event(MOUSEEVENTF_LEFTDOWN,0,0,0,0);
            Sleep(0.000001);
            mouse_event(MOUSEEVENTF_LEFTUP,0,0,0,0);  
        }  
        Sleep(0.000001);    
    }    
    return 0;  
}

长按版

#include<cstdio>  
#include<windows.h>    
#define KEY_DOWN(VK_NONAME) ((GetAsyncKeyState(VK_NONAME) & 0x8000) ? 1:0)   
int main()
{ 
    int now=0,k=0;    
    printf("将鼠标放在“采集猫薄荷”上，按1点击左键，按2停止\n");  
    while(1){    
        if(KEY_DOWN('1')){   
            now=1;     
            Sleep(100);   
        }
        if(KEY_DOWN('2')){    
            now=0;    
            Sleep(100);
        }
        if(now==1){   
            mouse_event(MOUSEEVENTF_LEFTDOWN,0,0,0,0);
            Sleep(1);
            mouse_event(MOUSEEVENTF_LEFTUP,0,0,0,0);  
        }  
        Sleep(1);    
    }    
    return 0;  
}
